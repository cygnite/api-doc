309:     /**
310:      * Resolve all dependencies of your class and return instance of
311:      * your class
312:      *
313:      * @param $class
314:      * @return mixed
315:      * @throws \Cygnite\Container\Exceptions\ContainerException
316:      */
317:     public function make($class, $arguments = [])
318:     {
319:         /*
320:          * If instance of the class already created and stored into
321:          * stack then simply return from here
322:          */
323:         if (isset($this[$class])) {
324:             return $this[$class];
325:         }
326: 
327:         $reflection = new Reflection();
328:         $reflectionClass = $reflection->setClass($class)->getReflectionClass();
329: 
330:         /*
331:          * Check if reflection class is not instantiable then throw ContainerException
332:          */
333:         if (!$reflectionClass->isInstantiable()) {
334:             $type = ($reflection->reflectionClass->isInterface() ? 'interface' : 'class');
335:             throw new ContainerException("Cannot instantiate $type $class");
336:         }
337: 
338:         $constructor = null;
339:         $constructorArgsCount = 0;
340:         if ($reflectionClass->hasMethod('__construct')) {
341:             $constructor = $reflectionClass->getConstructor();
342:             $constructor->setAccessible(true);
343:             $constructorArgsCount = $constructor->getNumberOfParameters();
344:         }
345: 
346:         // if class does not have explicitly defined constructor or constructor does not have parameters
347:         // get the new instance
348:         if (!isset($constructor) && is_null($constructor) || $constructorArgsCount < 1) {
349:             return $this[$class] = $reflectionClass->newInstance();
350:         }
351: 
352:         $dependencies = $constructor->getParameters();
353:         $constructorArgs = [];
354: 
355:         foreach ($dependencies as $dependency) {
356: 
357:             if (!is_null($dependency->getClass())) {
358:                 $constructorArgs[] = $this->resolverClass($dependency, $arguments);
359:             } else {
360:                 /*
361:                  | Check if construct has default value
362:                  | if exists we will simply assign it into array and continue
363:                  | for next argument
364:                  */
365:                 if ($dependency->isDefaultValueAvailable()) {
366: 
367:                     $constructorArgs[] = $this->checkIfConstructorHasDefaultArgs($dependency, $arguments);
368:                     continue;
369:                 }
370: 
371:                 /*
372:                  | Check parameters are optional or not
373:                  | if it is optional we will set the default value
374:                  */
375:                 $constructorArgs[] = $this->isOptionalArgs($dependency);
376:             }
377:         }
378: 
379:         return $this[$class] = $reflectionClass->newInstanceArgs($constructorArgs);
380:     }
381: 
382:     /**
383:      * @param $dependency
384:      * @param $arguments
385:      * @return array|mixed
386:      */
387:     private function resolverClass($dependency, $arguments)
388:     {
389:         list($resolveClass, $reflectionParam) = $this->getReflectionParam($dependency);
390: 
391:         // Application and Container cannot be injected into controller currently
392:         // since Application constructor is protected
393:         if ($reflectionParam->IsInstantiable()) {
394:             return $this->makeInstance($resolveClass, $arguments);
395:         }
396: 
397:         return $this->interfaceInjection($reflectionParam);
398:     }
