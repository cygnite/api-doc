192:     /**
193:      * We will create table schema here
194:      *
195:      * @param        $columns
196:      * @param string $engine
197:      * @param string $charset
198:      * @return $this
199:      */
200:     public function create($columns, $engine = 'MyISAM', $charset = 'utf8')
201:     {
202:         $schema = $comma = $isNull = $tableKey = $type = "";
203: 
204:         $schema .= strtoupper(__FUNCTION__) . ' TABLE IF NOT EXISTS
205:         `' . $this->database . '`.`' . $this->tableName . '` (';
206:         $arrCount = count($columns);
207:         $i = 0;
208: 
209:         foreach ($columns as $key => $value) {
210:             $increment = (isset($value['increment'])) ? 'AUTO_INCREMENT' : '';
211:             if (isset($value['key'])) {
212:                 $tableKey = strtoupper($value['key']) . ' ';
213: 
214:                 if ($value['key'] == 'primary') {
215:                     $tableKey = 'PRIMARY KEY';
216:                 }
217:             } else {
218:                 $tableKey = '';
219:             }
220: 
221:             $isNull = (isset($value['null']) &&
222:                 $value['null'] == true
223:             ) ? ' DEFAULT NULL' : ' NOT NULL';
224: 
225:             switch (strtolower($value['type'])) {
226: 
227:                 case 'int':
228:                     $len = (isset($value['length'])) ? $value['length'] : 11;
229:                     list($type, $length) = $this->columnType($value['type'], $len);
230:                     break;
231:                 case 'char':
232:                     list($type, $length) = $this->columnType($value['type'], $value['length'], 2);
233:                     break;
234:                 case 'string':
235:                     list($type, $length) = $this->columnType('varchar', $value['length'], 200);
236:                     break;
237:                 case 'text':
238:                     $type = 'TEXT';
239:                     break;
240:                 case 'longtext':
241:                     $type = $value['type'];
242:                     break;
243:                 case 'float':
244:                     list($type, $length) = $this->columnType($value['type'], $value['length'], '10,2');
245:                     break;
246:                 case 'decimal':
247:                     list($type, $length) = $this->columnType($value['type'], $value['length'], '10,2');
248:                     break;
249:                 case 'enum':
250:                     $length = implode('","', $value['length']);
251:                     $type = strtoupper($value['type']).'("'.$length.'")';
252:                     break;
253:                 case 'date':
254:                     $type = 'date';
255:                     break;
256:                 case 'datetime':
257:                     $len = (isset($value['length'])) ? $value['length'] : "NOT NULL DEFAULT '0000:00:00 00:00:00'";
258:                     $type = strtoupper($value['type']) . ' ' . $len;
259:                     $isNull = '';
260:                     break;
261:                 case 'time':
262:                     $type = 'time';
263:                     break;
264:                 case 'timestamp':
265:                     $len = (isset($value['length'])) ? $value['length'] : '';
266:                     $type = strtoupper($value['type']) . ' ' . $len;
267:                     break;
268:             }
269: 
270:             $comma = ($i < $arrCount - 1) ? ',' : '';
271: 
272:             $schema .= '`' . $value['column'] . "` " . strtoupper($type) . " "
273:                 . $increment . " " . $tableKey . PHP_EOL . $isNull . PHP_EOL;
274:             $schema .= $comma;
275: 
276:             $i++;
277:         }
278: 
279:         $schema .= ') ENGINE=' . $engine . ' DEFAULT  CHARSET=' . $charset . ';' . PHP_EOL;
280: 
281:         $this->schema = $schema;
282: 
283:         return $this;
284:     }
